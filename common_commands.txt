# some common commands used during debugging

# env cleaning
docker stop $(docker ps -q)
docker rm $(docker ps -aq)
docker rmi $(docker images --filter "dangling=true" -q --no-trunc)

# image build
docker build -t python-2-fixed .
docker build --no-cache -t python-2-fixed .

# run container
docker run --rm -it python-2-fixed bash
